const client = new PlainClient({
  apiKey: 'XXX',
});

const res = await client.getCustomers({
  first: 50,
  filters: { isMarkedAsSpam: false },
  sortBy: { field: CustomersSortField.FullName, direction: SortDirection.Asc },
});

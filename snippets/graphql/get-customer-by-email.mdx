```graphql filename="Query"
query getCustomerByEmail($email: String!) {
  customerByEmail(email: $email) {
    id
    fullName
    shortName
    email {
      email
      isVerified
    }
    externalId
    markedAsSpamAt {
      iso8601
    }
  }
}
```

}

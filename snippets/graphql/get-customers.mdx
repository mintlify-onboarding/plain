```graphql filename="Query"
query getCustomers($cursor: String!) {
  customers(
    after: $cursor
    first: 50
    filters: { isMarkedAsSpam: false }
    sortBy: { field: FULL_NAME, direction: ASC }
  ) {
    edges {
      node {
        id
        fullName
        shortName
        email {
          email
          isVerified
        }
        externalId
        markedAsSpamAt {
          iso8601
        }
      }
    }
    totalCount
    pageInfo {
      hasPreviousPage
      hasNextPage
      startCursor
      endCursor
    }
  }
}
```
